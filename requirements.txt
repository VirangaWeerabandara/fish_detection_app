# ── Computer Vision & ML ─────────────────────────────────────────────────────
#
# ⚠️  DO NOT install opencv-python on Jetson!
#     pip-opencv bundles its own Qt libraries which conflict with PyQt5
#     (crash: "Could not load Qt platform plugin xcb").
#     It also has NO GStreamer support, so the CSI camera won't work.
#
#     On Jetson, use the system OpenCV that ships with JetPack:
#       1. Create the venv with:  python3 -m venv --system-site-packages fish_venv
#       2. Do NOT install opencv-python / opencv-python-headless here.
#       3. Verify GStreamer is available:
#            python3 -c "import cv2; print(cv2.getBuildInformation())" | grep GStreamer
#            # Expected: GStreamer:   YES (1.x.y)
#
#     On a development PC (non-Jetson), uncomment the line below:
# opencv-python-headless   # headless avoids the Qt-bundle conflict with PyQt5

ultralytics   # 8.1+ required for YOLOv11 (C3k2 block)
torch         # 2.2+ required for macOS MPS stability + ultralytics 8.4
torchvision   # must match torch major.minor version
lap           # required by ByteTrack tracker (ultralytics dependency)
